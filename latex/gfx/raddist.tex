	\foreach \x in {-10,...,10} {

	\def\rr{(\xx*\xx+\yy*\yy)}

	\def\corr{(1.0+\ka*\rr+\kb*\rr*\rr)}
	\def\xd{\xx*\corr}
	\def\yd{\yy*\corr}

	%\draw ({\xx},{\yy}) -- ({\xd},{\yd});
	\draw [-stealth] ({\xx},{\yy}) -- ({\xd},{\yd});
	}
}
